LocalSDK Integration Guide

Requirements

System Requirements
  • Android API Level 21+ (Android 5.0)
  • Kotlin or Java project
  • Android Gradle Plugin 7.0+
  • Target SDK 34 (Android 14) recommended

Installation

Add Maven Central Repository

// In your app-level build.gradle.kts
repositories {
    mavenCentral()
}

Add Dependency

dependencies {
    implementation("com.localsdk:core:0.0.1-alpha")
}

Permissions

Add the required permissions to your AndroidManifest.xml:

<!-- Required location permissions -->
<uses-permission android:name="android.permission.ACCESS_FINE_LOCATION" />
<uses-permission android:name="android.permission.ACCESS_COARSE_LOCATION" />

<!-- Background location (Android 10+) -->
<uses-permission android:name="android.permission.ACCESS_BACKGROUND_LOCATION" />

<!-- Foreground service -->
<uses-permission android:name="android.permission.FOREGROUND_SERVICE" />
<uses-permission android:name="android.permission.FOREGROUND_SERVICE_LOCATION" />

<!-- Battery optimization bypass -->
<uses-permission android:name="android.permission.REQUEST_IGNORE_BATTERY_OPTIMIZATIONS" />
Important: Background location permission requires additional steps on Android 10+. Request it separately after obtaining foreground location permission.

SDK Key Setup

Required: You must obtain an SDK key to use LocalSDK. Contact your LocalSDK representative or visit our website to get your unique SDK key.

Add SDK Key to Manifest

Add your SDK key to your AndroidManifest.xml in the <application> section:

<application>
    <meta-data
        android:name="com.localsdk.API_KEY"
        android:value="YOUR_SDK_KEY_HERE" />
    ...
</application>

Alternative: Programmatic Setup

You can also initialize the SDK programmatically with your context and SDK key:

LocalSDK.initialize(this, "YOUR_SDK_KEY_HERE")

Basic Setup

Initialize SDK

import com.localsdk.LocalSDK
import com.localsdk.config.Config
import com.localsdk.config.TrackingMode

class MainActivity : AppCompatActivity() {
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)

        // Initialize SDK with key (if not in manifest)
        LocalSDK.initialize(this, "YOUR_SDK_KEY_HERE")

        // Configure SDK
        val config = Config.Builder()
            .setTrackingMode(TrackingMode.STANDARD)
            .setForegroundServiceEnabled(true)
            .setUpdateIntervalMs(5000) // 5 seconds
            .build()

        LocalSDK.setConfig(config)
    }
}

Permission Handling

import com.localsdk.permissions.LocationPermissions

private fun setupLocationPermissions() {
    // Check if permissions are granted
    if (!LocationPermissions.checkLocationPermission(this)) {
        // Request location permission
        LocationPermissions.requestLocationPermission(this) { granted ->
            if (granted) {
                startLocationTracking()
            } else {
                // Handle permission denied
                showPermissionDeniedDialog()
            }
        }
    } else {
        startLocationTracking()
    }
}

// For background location (call after foreground permission granted)
private fun requestBackgroundLocationPermission() {
    if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.Q) {
        if (!LocationPermissions.checkBackgroundLocationPermission(this)) {
            LocationPermissions.requestBackgroundLocationPermission(this)
        }
    }
}

Start Tracking

private fun startLocationTracking() {
    // Set up location callback
    LocalSDK.onLocation { location ->
        // Handle location update
        Log.d("Location", "Lat: ${location.latitude}, Lng: ${location.longitude}")
        updateUI(location)
    }

    // Set up error callback
    LocalSDK.onError { error ->
        Log.e("LocalSDK", "Error: ${error.message}")
        handleLocationError(error)
    }

    // Start tracking
    LocalSDK.startTracking()
}

// Stop tracking when needed
private fun stopLocationTracking() {
    LocalSDK.stopTracking()
}

Advanced Configuration

Custom Configuration

val config = Config.Builder()
    .setTrackingMode(TrackingMode.HIGH_ACCURACY)
    .setUpdateIntervalMs(10000) // 10 seconds
    .setFastestIntervalMs(5000)  // 5 seconds
    .setSmallestDisplacementMeters(10.0f)
    .setForegroundServiceEnabled(true)
    .setNotificationTitle("Location Tracking")
    .setNotificationText("App is tracking your location")
    .build()

LocalSDK.setConfig(config)

Tracking Modes

  • TrackingMode.HIGH_ACCURACY - GPS + Network (high battery usage)
  • TrackingMode.STANDARD - Balanced accuracy and battery (recommended)
  • TrackingMode.LOW_POWER - Network only (low battery usage)

Battery Optimization

import com.localsdk.permissions.BatteryOptimizations

private fun handleBatteryOptimization() {
    if (!BatteryOptimizations.isIgnoringBatteryOptimizations(this)) {
        // Show dialog explaining why this is needed
        showBatteryOptimizationDialog {
            BatteryOptimizations.requestIgnoreBatteryOptimizations(this)
        }
    }
}

Complete Example

class MainActivity : AppCompatActivity() {
    private lateinit var locationText: TextView

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)

        locationText = findViewById(R.id.locationText)

        setupSDK()
        setupPermissions()
    }

    private fun setupSDK() {
        // Initialize SDK with key (if not in manifest)
        LocalSDK.initialize(this, "YOUR_SDK_KEY_HERE")

        val config = Config.Builder()
            .setTrackingMode(TrackingMode.STANDARD)
            .setForegroundServiceEnabled(true)
            .setUpdateIntervalMs(10000)
            .build()

        LocalSDK.setConfig(config)
    }

    private fun setupPermissions() {
        if (LocationPermissions.checkLocationPermission(this)) {
            startTracking()
        } else {
            LocationPermissions.requestLocationPermission(this) { granted ->
                if (granted) startTracking()
            }
        }
    }

    private fun startTracking() {
        LocalSDK.onLocation { location ->
            runOnUiThread {
                locationText.text = "Lat: ${location.latitude}\nLng: ${location.longitude}"
            }
        }

        LocalSDK.onError { error ->
            Log.e("LocalSDK", "Error: ${error.message}", error.cause)
        }

        LocalSDK.startTracking()
    }

    override fun onDestroy() {
        super.onDestroy()
        LocalSDK.stopTracking()
    }
}
